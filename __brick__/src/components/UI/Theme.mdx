import { Meta, ColorPalette, ColorItem, Unstyled } from '@storybook/blocks';
import resolveConfig from 'tailwindcss/resolveConfig'
import tailwindConfig from '../../../tailwind.config.ts'

<Meta title="Foundations / Theme" />

# Theme

## Colors
<Unstyled>
  <ColorPalette>
    <div class='flex flex-col gap-8 mb-20'>
      {Object.entries(resolveConfig(tailwindConfig).theme.colors).map(([key, value]) => {
        if (typeof value === 'string') {
          return (
            <ColorItem
              title={key}
              colors={{ [key]: value }}
            />
          )
        } else {
          return (
            <div>
              <p class='font-bold text-lg mb-2'>
                {key}
              </p>

              <div class='border border-border rounded-2xl p-8'>
                {Object.entries(value).map(([subkey, subvalue]) => {
                  return (
                    <ColorItem
                      title={`${key}-${subkey}`}
                      colors={{ [subkey]: subvalue }}
                    />
                  )
                })}
              </div>
            </div>
          )
        }
      })}
    </div>
  </ColorPalette>
</Unstyled>

## Radius

  </div>
</Unstyled>
